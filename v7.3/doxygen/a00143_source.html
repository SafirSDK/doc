315 void Copy(const ContainerBase& that) override

359 void Merge(const DictionaryContainerBase& that) override
360 {
361 const DictionaryContainer<KeyT, ValT>& other = Cast(that);
362
363 for (const_iterator it = other.begin(); it != other.end(); ++it)
364 {
365 const ObjectContainerBase* fromContainerOB =
366 dynamic_cast<const ObjectContainerBase*>(&it->second);
367 //is it an object member?
368 if (fromContainerOB != NULL)
369 {
370 if (fromContainerOB->IsChangedHere())
371 {
372 iterator findIt = find(it->first);
373 if (findIt == end())
374 {
375 throw SoftwareViolationException
376 (L"DictionaryContainer::Merge: Changed key not found in target!",
377 __WFILE__, __LINE__);
378 }
379
380 findIt->second.Copy(*fromContainerOB);
381 }
382 else if (fromContainerOB->IsChanged())
383 {
384 iterator findIt = find(it->first);
385 if (findIt == end())
386 {
387 throw SoftwareViolationException
388 (L"DictionaryContainer::Merge: Changed key not found in target!",
389 __WFILE__, __LINE__);
390 }
391 ObjectContainerBase* intoContainerOB = dynamic_cast<ObjectContainerBase*>(&findIt->second);
392
393 ObjectPtr into = intoContainerOB->GetObjectPointer();
394 ObjectConstPtr from = fromContainerOB->GetObjectPointer();
395
396 //recurse
397 Utilities::MergeChanges(into,from);
398 }
399 }
400 else
401 {
402 if (it->second.IsChanged())
403 {
404 iterator findIt = find(it->first);
405 if (findIt == end())
406 {
407 throw SoftwareViolationException
408 (L"DictionaryContainer::Merge: Changed key not found in target!",
409 __WFILE__, __LINE__);
410 }
411
412 findIt->second.Copy(it->second);
413 }
414 }
415 }
