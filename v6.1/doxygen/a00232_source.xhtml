200  virtual void Copy(const ContainerBase& that)
201  {
202  if (this != &that)
203  {
204  if (typeid(*this) != typeid(that))
205  {
206  throw SoftwareViolationException(L"Invalid call to Copy, containers are not of same type",__WFILE__,__LINE__);
207  }
208 
209  const SequenceContainer<ContainedType>& other=static_cast<const SequenceContainer<ContainedType>& >(that);
210 
211  m_bIsChanged=other.m_bIsChanged;
212  m_values.clear();
213  m_values.reserve(other.m_values.size());
214  for (typename StorageType::const_iterator it=other.m_values.begin(); it!=other.m_values.end(); ++it)
215  {
216  m_values.push_back(SequenceCopyHelper<ContainedType>::Copy(*it));
217  }
218  }
219  }
