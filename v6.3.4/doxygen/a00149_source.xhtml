209  throw SoftwareViolationException(L"Invalid call to Copy, containers are not of same type",__WFILE__,__LINE__);

290  template <class T>
291  class GenericObjectSequenceContainer
292  : public SequenceContainerBase<boost::shared_ptr<T> >
293  , public GenericObjectSequenceContainerBase
294  {
295  typedef SequenceContainerBase<boost::shared_ptr<T> > Base;
296  public:
297 
298  //Override of inherited method. Parent comment describes this behaviour too..
299  virtual bool IsChanged() const
300  {
301  if (Base::m_bIsChanged)
302  {
303  return true;
304  }
305 
306  for (typename Base::const_iterator it=Base::begin(); it!=Base::end(); ++it)
307  {
308  if ((*it)->IsChanged()) //something in an object has changed
309  {
310  return true;
311  }
312  }
313 
314  return false;
315  }
316 
324  bool IsChangedHere() const
325  {
326  return Base::m_bIsChanged;
327  }
328 
329  //Override of inherited method. Parent comment describes this behaviour too..
330  virtual void SetChanged(const bool changed)
331  {
332  Base::m_bIsChanged = changed;
333 
334  for (typename Base::const_iterator it=Base::begin(); it!=Base::end(); ++it)
335  {
336  (*it)->SetChanged(changed);
337  }
338  }
339 
347  void SetChangedHere(const bool changed)
348  {
349  Base::m_bIsChanged = changed;
350  }
351 
352  private:
353  virtual void Merge(const GenericObjectSequenceContainerBase& that)
354  {
355 #ifndef NDEBUG
356  if (typeid(GenericObjectSequenceContainer<T>) != typeid(that))
357  {
358  throw SoftwareViolationException(L"Invalid call to Merge, containers are not of same type",
359  __WFILE__,__LINE__);
360  }
361 #endif
362 
363  const GenericObjectSequenceContainer<T>& other =
364  static_cast<const GenericObjectSequenceContainer<T>&>(that);
365 
366  //Note: this function only gets called when IsChangedHere() == false
367 
368  if (!other.IsChanged())
369  {
370  return;
371  }
372 
373  if (Base::size() != other.size())
374  {
375  throw SoftwareViolationException(L"It is not possible to merge two object sequences of different sizes.",
376  __WFILE__,__LINE__);
377  }
378 
379  for (size_t i = 0; i < Base::size(); ++i)
380  {
381  if (other.at(i)->IsChanged())
382  {
383  //recurse
384  Utilities::MergeChanges(Base::at(i),other.at(i));
385  }
386  }
387  }
